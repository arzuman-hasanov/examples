Rəqəmlərdən ibarət array qəbul edib array-də olan rəqəmlərin ortalamasını qaytaran bir funksiya yaz.

            let array = [1,2,3,4,5,6,7];
            let sum = 0;
            let result;
            
            for(let i = 0 ; i < array.length ; i++) {
                sum += array[i];
                if(i == array.length - 1){
                    result = sum / (i+1);
                }
            }

            console.log("rəqəmlərin ortalaması: " + result);
